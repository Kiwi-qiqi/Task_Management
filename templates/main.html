Multi-Page Navigation System

This demo shows how to navigate between multiple HTML pages using a sidebar. Click on the navigation items in the sidebar to see the page switching effect. The currently active page is highlighted in the sidebar.

Implementation Approach

For this to work across multiple HTML files:

  1. Copy the exact same sidebar HTML to all pages
  2. Use relative paths in href attributes
  3. Add a unique data-page attribute to each navigation link
  4. Use the JavaScript code below in all pages
<script>
// Page load: Activate correct nav item
document.addEventListener('DOMContentLoaded', function() {
  const currentPage = window.location.pathname.split('/').pop().split('.')[0];
  const navLinks = document.querySelectorAll('.nav-link');

  navLinks.forEach(link => {
    link.classList.remove('active');
    if (link.getAttribute('data-page') === currentPage) {
      link.classList.add('active');
    }
  });
});

// Click events for navigation
document.querySelectorAll('.nav-link').forEach(link => {
  if (link.id !== 'logoutBtn') {
    link.addEventListener('click', function(e) {
      e.preventDefault();
      // Store active page in session for multi-page apps
      sessionStorage.setItem('lastActiveLink', this.getAttribute('data-page'));
      window.location.href = this.href;
    });
  }
});
// Logout functionality
document.getElementById('logoutBtn').addEventListener('click', function(e) {
  e.preventDefault();
  // Clear authentication state
  localStorage.removeItem('isLoggedIn');
  // Redirect to login page
  window.location.href = 'login.html';
});
</script>
